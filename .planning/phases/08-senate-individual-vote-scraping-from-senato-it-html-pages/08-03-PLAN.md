def parse_senate_votazione_uri(uri: str):
    m = _SEN_VOTAZIONE_URI_RE.search(uri)
    if not m:
        return None, None, None
    return int(m.group(1)), int(m.group(2)), int(m.group(3))  # (leg, seduta, vote)

def _derive_senate_outcome(favorevoli: int, maggioranza: int) -> str:
    if favorevoli is None or maggioranza is None:
        return "unknown"
    return "approved" if favorevoli >= maggioranza else "rejected"
