```python
from pydantic import BaseModel
from typing import Optional

class DebateSummary(BaseModel):
    id: str
    title: str
    speech_count: int

class SessionCard(BaseModel):
    id: str
    date: str                         # ISO date string "2026-04-07"
    chamber: str                      # "camera" or "senato"
    number: int
    recap: Optional[str] = None       # None if not yet generated
    debate_count: int
    vote_count: int
    speech_count: int
    debates: list[DebateSummary]

class TimelineResponse(BaseModel):
    sessions: list[SessionCard]
    next_cursor: Optional[str] = None # ISO date of last session
    has_more: bool

class PhaseInfo(BaseModel):
    id: str
    title: str
    phase_type: Optional[str] = None
    speech_count: int

class VoteInfo(BaseModel):
    id: str
    number: int
    subject: Optional[str] = None
    outcome: Optional[str] = None
    in_favor: Optional[int] = None
    against: Optional[int] = None
    abstained: Optional[int] = None

class ActInfo(BaseModel):
    id: str
    title: Optional[str] = None
    type: Optional[str] = None

class SpeakerInfo(BaseModel):
    id: str
    first_name: str
    last_name: str
    party: Optional[str] = None
    speaking_role: Optional[str] = None
    is_government_member: bool = False
    speech_count: int
    phases: list[str]               # phase titles where speaker participated

class DebateDetailResponse(BaseModel):
    id: str
    title: str
    recap: Optional[str] = None
    phases: list[PhaseInfo]
    speakers: list[SpeakerInfo]
    votes: list[VoteInfo]
    acts: list[ActInfo]

class SpeakerSummaryResponse(BaseModel):
    summary: Optional[str] = None
    speech_count: int
    phases: list[str]
```

**Session list Cypher query:**
```cypher
MATCH (s:Session)
WHERE ($chambers IS NULL OR s.chamber IN $chambers)
  AND ($before IS NULL OR s.date < date($before))
  AND ($from_date IS NULL OR s.date >= date($from_date))
  AND ($to_date IS NULL OR s.date <= date($to_date))
WITH s ORDER BY s.date DESC LIMIT $limit + 1
OPTIONAL MATCH (s)-[:HAS_DEBATE]->(d)
OPTIONAL MATCH (d)-[:HAS_PHASE]->(p)-[:CONTAINS_SPEECH]->(sp)
WITH s, d, count(DISTINCT sp) AS dSpeechCount
WITH s, collect(DISTINCT {id: d.id, title: d.title, speechCount: dSpeechCount}) AS debates
OPTIONAL MATCH (s)-[:HAS_VOTE]->(v)
WITH s, debates, count(DISTINCT v) AS voteCount
RETURN s.id AS id, toString(s.date) AS date, s.chamber AS chamber, s.number AS number,
       s.recapIt AS recapIt, s.recapEn AS recapEn,
       debates, voteCount,
       size(debates) AS debateCount,
       reduce(total = 0, d IN debates | total + d.speechCount) AS speechCount
ORDER BY s.date DESC
```

- Convert `chamber` param: "both" -> `None` (no filter), "camera" -> `["camera"]`, "senato" -> `["senato"]`
- Select `recapIt` or `recapEn` based on locale
- Fetch `limit + 1` rows; if got > limit, set `has_more = True` and `next_cursor = last_session.date`; return only first `limit` rows
- For search: add `AND (toLower(d.title) CONTAINS toLower($search) OR toLower(s.recapIt) CONTAINS toLower($search) OR toLower(s.recapEn) CONTAINS toLower($search))` to WHERE clause. Note: search filtering happens at the debate level but results are session-level, so a session matches if ANY of its debates match.

**File 1: `backend/app/routers/timeline.py`**

Create thin router with three endpoints:

```python
"""
Timeline endpoints for browsing parliamentary sessions, debates, and speaker summaries.

Read-only endpoints serving pre-computed AI summaries from Neo4j.
"""
from typing import Optional

from fastapi import APIRouter, Depends, Request

from app.models.timeline import (
    DebateDetailResponse,
    SpeakerSummaryResponse,
    TimelineResponse,
)
from app.services.deps import get_neo4j_client
from app.services.neo4j_client import Neo4jClient
from app.services import timeline_service

router = APIRouter(prefix="/api/timeline", tags=["timeline"])


@router.get("")
async def get_timeline(
    request: Request,
    before: Optional[str] = None,
    limit: int = 20,
    chamber: str = "both",
    search: Optional[str] = None,
    from_date: Optional[str] = None,
    to_date: Optional[str] = None,
    neo4j: Neo4jClient = Depends(get_neo4j_client),
) -> TimelineResponse:
    """Get paginated timeline of parliamentary sessions."""
    locale = request.headers.get("Accept-Language", "it")[:2]
    return await timeline_service.get_sessions(
        neo4j=neo4j, locale=locale, before=before, limit=limit,
        chamber=chamber, search=search, from_date=from_date, to_date=to_date,
    )


@router.get("/debates/{debate_id}")
async def get_debate_detail(
    debate_id: str,
    request: Request,
    neo4j: Neo4jClient = Depends(get_neo4j_client),
) -> DebateDetailResponse:
    """Get detailed debate information with phases, speakers, votes, and acts."""
    locale = request.headers.get("Accept-Language", "it")[:2]
    return await timeline_service.get_debate_detail(
        neo4j=neo4j, debate_id=debate_id, locale=locale,
    )


@router.get("/speakers/{debate_id}/{speaker_id}")
async def get_speaker_summary(
    debate_id: str,
    speaker_id: str,
    request: Request,
    neo4j: Neo4jClient = Depends(get_neo4j_client),
) -> SpeakerSummaryResponse:
    """Get AI-generated speaker position summary for a specific debate."""
    locale = request.headers.get("Accept-Language", "it")[:2]
    return await timeline_service.get_speaker_summary(
        neo4j=neo4j, debate_id=debate_id, speaker_id=speaker_id, locale=locale,
    )
```
