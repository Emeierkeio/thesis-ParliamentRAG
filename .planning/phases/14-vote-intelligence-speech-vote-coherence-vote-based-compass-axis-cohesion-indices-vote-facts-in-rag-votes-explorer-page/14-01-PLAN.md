    ```python
    def rice_index(favor: int, against: int) -> float | None:
        """Rice cohesion for one party on one vote. |F-A|/(F+A). None if F+A==0 (abstain-only)."""
        denom = favor + against
        if denom == 0:
            return None
        return abs(favor - against) / denom

    def mean_rice(vote_rows: list[dict], exclude_unanimous: bool = True) -> float:
        """Mean Rice across votes. vote_rows = [{'favor': int, 'against': int}, ...].
        Excludes votes where rice==1.0 when exclude_unanimous=True (procedural unanimity inflates cohesion)."""
        indices = [rice_index(r["favor"], r["against"]) for r in vote_rows]
        indices = [r for r in indices if r is not None]
        if exclude_unanimous:
            indices = [r for r in indices if r < 1.0]
        return sum(indices) / len(indices) if indices else 0.0
    ```
