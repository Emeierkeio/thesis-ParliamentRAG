    Create `backend/app/routers/votes.py`:

    ```python
    from fastapi import APIRouter, Depends
    from ..services.deps import get_neo4j_client
    from ..services.neo4j_client import Neo4jClient
    from ..services import votes_service

    router = APIRouter(prefix="/api", tags=["votes"])

    @router.get("/votes")
    async def search_votes_endpoint(
        chamber: str = "both",
        legislature: int = 19,
        from_date: str | None = None,
        to_date: str | None = None,
        outcome: str | None = None,
        min_margin: int | None = None,
        limit: int = 50,
        offset: int = 0,
        neo4j: Neo4jClient = Depends(get_neo4j_client),
    ) -> dict:
        return votes_service.search_votes(
            neo4j, chamber=chamber, legislature=legislature, from_date=from_date,
            to_date=to_date, outcome=outcome, min_margin=min_margin, limit=limit, offset=offset,
        )

    @router.get("/rankings/votes")
    async def rankings_votes_endpoint(
        chamber: str = "camera",
        legislature: int = 19,
        deputy_id: str | None = None,
        neo4j: Neo4jClient = Depends(get_neo4j_client),
    ) -> dict:
        if deputy_id:
            return votes_service.get_deputy_vote_stats(neo4j, deputy_id, chamber, legislature)
        return votes_service.get_party_cohesion(neo4j, chamber, legislature)
    ```

    In `backend/app/main.py`: add `from .routers.votes import router as votes_router` to the router import block, and `app.include_router(votes_router)  # Vote intelligence: search + cohesion rankings` to the include block.
