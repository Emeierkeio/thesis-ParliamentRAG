   ```typescript
   import { getRequestConfig } from 'next-intl/server';
   import { cookies } from 'next/headers';

   const VALID_LOCALES = ['it', 'en'] as const;
   type Locale = (typeof VALID_LOCALES)[number];

   export default getRequestConfig(async () => {
     const cookieStore = await cookies();
     const raw = cookieStore.get('NEXT_LOCALE')?.value ?? 'it';
     const locale: Locale = VALID_LOCALES.includes(raw as Locale) ? (raw as Locale) : 'it';

     return {
       locale,
       messages: (await import(`../../messages/${locale}.json`)).default,
     };
   });
   ```

1. Create `backend/app/services/translation.py`:
   - Import `asyncio`, `json`, `logging` and `make_async_client` from `..key_pool`
   - Define `TRANSLATION_PROMPT` as a module-level constant:
     ```
     Translate the following Italian parliamentary speech excerpts to English.
     Preserve formal parliamentary register. Do not translate proper nouns
     (speaker names, party names, place names, dates, session numbers).
     Return ONLY valid JSON: {"text": "...", "full_text": "..."}

     text: {text}
     full_text: {full_text}
     ```
   - Define `async def translate_citation_batch(citations: list[dict], target_lang: str = "en") -> list[dict]`:
     - If target_lang == "it" or not citations, return citations unchanged
     - Create async client via `make_async_client()`
     - Use `asyncio.gather(*[_translate_one(c, client) for c in citations], return_exceptions=True)`
     - For any result that is an Exception, replace with original citation (log warning)
     - Return list of translated citations
   - Define `async def _translate_one(citation: dict, client) -> dict`:
     - Extract text = citation.get("text", "") and full_text = citation.get("full_text", "")
     - If not text and not full_text, return citation unchanged
     - Build prompt from TRANSLATION_PROMPT.format(text=text, full_text=full_text)
     - Call `client.chat.completions.create(model="gpt-4o-mini", messages=[{"role": "user", "content": prompt}], temperature=0)`
     - Parse JSON from response.choices[0].message.content
     - Return `{**citation, "translated_text": result.get("text", ""), "translated_full_text": result.get("full_text", ""), "is_translated": True}`
     - On ANY exception: log warning, return citation unchanged (no translated_* keys)
